30 {
31 type::Int32 seed = 131;
32 type::Int32 hash = 0;
33 for (int i = 0; (len == -1 || i < len) && str[i]; ++i) {
34 hash = static_cast<type::Int32>(static_cast<type::Int64>(hash) * seed + str[i]);
35 }
36 return hash & 0x7fffffff;
37 }

39 static constexpr type::UInt32 HexStrToUInt32(const char* str, int len = -1)
40 {
41 type::UInt32 ans = 0;
42 for (int i = 0; (len == -1 || i < len) && str[i]; ++i) {
43 ans <<= 4;
44 int tmp = 0;
45 if (str[i] >= 'a' && str[i] <= 'f') {
46 tmp = str[i] - 'a' + 10;
47 } else if (str[i] >= 'A' && str[i] <= 'F') {
48 tmp = str[i] - 'A' + 10;
49 } else if (str[i] >= '0' && str[i] <= '9') {
50 tmp = str[i] - '0';
51 } else {
52 CHERRYSODA_ASSERT(false, "Hex string invalid.\n");
53 }
54 ans |= tmp;
55 }
56 return ans;
57 }
