66 static constexpr type::UInt32 EncodeNormalU32(const Math::Vec3& v)
67 {
68 type::UInt32 a = static_cast<type::UInt32>((v.x + 1.0f) * 0.5f * 255.f + 0.5f);
69 type::UInt32 b = static_cast<type::UInt32>((v.y + 1.0f) * 0.5f * 255.f + 0.5f);
70 type::UInt32 c = static_cast<type::UInt32>((v.z + 1.0f) * 0.5f * 255.f + 0.5f);
71 return a | b << 8 | c << 16;
72 }
